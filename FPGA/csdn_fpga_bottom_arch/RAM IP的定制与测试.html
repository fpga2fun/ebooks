2、RAM IP的例化与仿真测试

2.1、例化一个RAM IP核

        按上述步骤生成IP后，复制 IP核 自带的例化模板。

        

        然后在我们的RTL代码中，例化生成一个4*16的SDP RAM。

        该RTL不做其他功能，只例化RAM即可。 

  1. module test(
  2. 	input 			clk,		//输入时钟
  3. 	input 	[3:0]	data_in,	//输入数据
  4. 	input			en_wr,		//写使能信号
  5. 	input			en_rd,		//读使能信号
  6. 	input	[3:0]	addr_wr,	//写地址		
  7. 	input	[2:0]	addr_rd,	//读地址
  8. 	input			we_wr,		//写使能	
  9. 	output	[7:0]	data_out	//输出数据
  10.     );
  11.  
  12. //例化RAM IP核
  13. ram_w4_d16	ram_w4_d16_inst (
  14. //端口A--写端口
  15. 	.clka	(clk),   
  16. 	.ena	(en_wr),    
  17. 	.wea	(we_wr),    
  18. 	.addra	(addr_wr),  
  19. 	.dina	(data_in),   
  20. //端口B--读端口  
  21. 	.clkb	(clk),   
  22. 	.enb	(en_rd),    
  23. 	.addrb	(addr_rd),  
  24. 	.doutb	(data_out)  
  25. );
  26. 	
  27. endmodule

2.2、仿真测试 

        写个testbench，对RAM IP测试一下。

        测试行为：先往地址0-15写入数据0-15，然后从地址0-7读出数据（因为读数据位宽是写数据位宽的2倍，所以读数据一次会读出2个写地址的数据）。

  1. `timescale 1ns / 1ns
  2.  
  3. module tb_test();
  4.  
  5. reg 			clk;		
  6. reg 	[3:0]	data_in;	
  7. reg				en_wr;			
  8. reg				en_rd;			
  9. reg		[3:0]	addr_wr;	
  10. reg		[2:0]	addr_rd;	
  11. reg				we_wr;		
  12. wire	[7:0]	data_out;	
  13.  
  14. initial begin
  15. 	clk = 0;
  16. 	en_wr = 0;en_rd = 0;we_wr = 0;
  17. 	data_in = 4'd0;
  18. 	addr_wr = 4'd0;
  19. 	addr_rd = 3'd0;
  20. 	
  21. 	#110	
  22. 	en_wr = 1;we_wr = 1;	//开始写入数据
  23. 	repeat(15) begin		//向地址0-15写入数据0-15
  24. 		#10
  25. 		data_in = data_in + 1;
  26. 		addr_wr = addr_wr + 1;		
  27. 	end
  28. 	
  29. 	#10
  30. 	en_wr = 0;				//停止写入数据
  31. 	en_rd = 1;we_wr = 0;	//开始读出数据
  32. 	repeat(10) begin		//从地址0-7读数据
  33. 		#10	addr_rd = addr_rd + 1;			
  34. 	end	
  35. 	en_rd = 0;				//停止读出数据
  36. 	#20 $finish;			//结束仿真	
  37. end
  39. always #5 clk = ~clk;		//生成时钟，周期10ns
  41. //例化被测模块test
  43. test	test_inst(
  44. 	.clk		(clk		),	
  45. 	.data_in	(data_in	),	
  46. 	.en_wr		(en_wr		),	
  47. 	.en_rd		(en_rd		),	
  48. 	.addr_wr	(addr_wr	),	
  49. 	.addr_rd	(addr_rd	),	
  50. 	.we_wr		(we_wr		),	
  51. 	.data_out	(data_out	)	
  52. );
  54. endmodule
